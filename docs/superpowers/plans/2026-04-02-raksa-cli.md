# Raksa CLI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a publishable Python CLI (`raksa`) for importing housing company renovation notifications and fault reports into EstateApp's GraphQL API.

**Architecture:** Thin GraphQL client (`client.py`) backed by Pydantic models (`models.py`), exposed through a Typer CLI with subcommand groups for renovations, faults, and auth. Import commands read hive-structured YAML directories (`year=*/house=*/*.yaml`) and transform them to EstateApp API inputs.

**Tech Stack:** Python 3.14, httpx, pydantic, typer, pyyaml, python-dotenv, uv, pytest

**Spec:** `docs/superpowers/specs/2026-04-02-estateapp-cli-design.md`

**User preferences:**
- `uv` for deps, `source activate && python -m pytest` for tests
- Modern type hints (`str | None`), early returns, specific exceptions
- tests/ mirrors src/, shared fixtures in conftest.py, no mocking unless strictly necessary
- Conventional commits, short first line + bullet points

import json
import os
from pathlib import Path

from dotenv import load_dotenv


CONFIG_DIR = Path.home() / ".config" / "raksa"


class RaksaConfigError(Exception):
    pass


def resolve_token() -> str:
    load_dotenv()

    token = os.environ.get("RAKSA_TOKEN")
    if token:
        return token

    token_file = CONFIG_DIR / "token.json"
    if token_file.exists():
        data = json.loads(token_file.read_text())
        token = data.get("loginToken")
        if token:
            return token

    raise RaksaConfigError(
        "No token found. Set RAKSA_TOKEN in .env, as an env var, "
        "or run 'raksa auth setup'."
    )


def resolve_condo_id() -> str | None:
    load_dotenv()
    return os.environ.get("RAKSA_CONDO_ID")


def resolve_base_url() -> str:
    load_dotenv()
    return os.environ.get("RAKSA_BASE_URL", "https://app.estateapp.com")


def save_token(login_token: str, user_id: str) -> Path:
    CONFIG_DIR.mkdir(parents=True, exist_ok=True)
    token_file = CONFIG_DIR / "token.json"
    token_file.write_text(json.dumps({"loginToken": login_token, "userId": user_id}, indent=2))
    return token_file
```

            self.request_type is not None
            and self.request_type.label == "Huoneistomuutosilmoitus"
        )

    def _parse_estimated_timing(self) -> tuple[str | None, str | None]:
        if not self.estimated_timing:
            return None, None
        match = re.match(r"(\d{2}\.\d{2}\.\d{4})\s*-\s*(\d{2}\.\d{2}\.\d{4})", self.estimated_timing)
        if not match:
            return None, None
        start_str, end_str = match.groups()
        start = datetime.datetime.strptime(start_str, "%d.%m.%Y")
        end = datetime.datetime.strptime(end_str, "%d.%m.%Y")
        return str(int(start.timestamp() * 1000)), str(int(end.timestamp() * 1000))

    def _map_chosen_jobs(self) -> dict[str, bool]:
        jobs: dict[str, bool] = {}
        for rt in self.renovation_types or []:
            if rt.type is None:
                continue
            field = RENOVATION_TYPE_MAP.get(rt.type.label, "otherChanges")
            jobs[field] = True
        return jobs

    def to_renovation_input(self, condo_id: str) -> dict:
        start_date, end_date = self._parse_estimated_timing()

        space_name = self.space.name if self.space else ""
        address = self.building_address or ""
        if space_name:
            address = f"{address} {space_name}".strip()

        contractors = []
        work_parts = []
        for rt in self.renovation_types or []:
            contractor_id = str(uuid.uuid4())
            contact_id = str(uuid.uuid4())
            contractors.append({
                "_id": contractor_id,
                "companyName": rt.organization_name,
                "companyBusinessId": rt.business_id,
                "contact": {
                    "_id": contact_id,
                    "name": rt.organization_name,
                    "phone": rt.mobile_phone,
                    "email": rt.email,
                },
            })
            if rt.description:
                work_parts.append(rt.description)

        work_description = self.public_description or ""
        if work_parts and not work_description:
            work_description = "\n\n".join(work_parts)

        informant_name = self.contact_person.name if self.contact_person else None
        informant_id = str(uuid.uuid4())

        return {
            "condominiumId": condo_id,
            "type": "shareholderRenovationWork",
            "status": "UPCOMING",
            "startDate": start_date,
            "endDate": end_date,
            "title": f"{self._job_summary()}, {space_name}" if space_name else self._job_summary(),
            "shareholderRenovationWork": {

                    "name": informant_name,

                    "understandContractorLiability": True,
                    "infoProvidedIsAccurate": True,
                    "acceptModificationTerms": True,
                    "awareOfProcessingAfterPayment": True,
                },
            },
        }

    def _job_summary(self) -> str:
        if not self.renovation_types:
            return "Muutostyö"
        labels = [rt.type.label for rt in self.renovation_types if rt.type]
        return ", ".join(labels) if labels else "Muutostyö"

            "apartment": self.space.name if self.space else None,
            "streetAddress": self.building_address,
            "contactName": self.contact_person.name if self.contact_person else None,
        }

class EstateAppClient:
    def __init__(
        self,
        token: str,
        base_url: str = "https://app.estateapp.com",
        http_client: httpx.Client | None = None,
    ):
        self._token = token
        self._base_url = base_url
        self._http = http_client or httpx.Client(timeout=60)

    def _gql(self, query: str, variables: dict | None = None) -> dict:
        headers = {
            "Content-Type": "application/json",
            "authorization": self._token,
        }
        resp = self._http.post(
            f"{self._base_url}/graphql",
            json={"query": query, "variables": variables or {}},
            headers=headers,
        )
        resp.raise_for_status()
        body = resp.json()
        if "errors" in body:
            messages = [e.get("message", "Unknown error") for e in body["errors"]]
            raise EstateAppError("; ".join(messages))
        return body.get("data", {})

        data = self._gql(LIST_RENOVATIONS_QUERY, {"condominiumId": condo_id})
        items = data.get("getCondominiumShareholderRenovationGigs") or []
        return [ShareholderRenovation.model_validate(item) for item in items]

        item = data.get("faultNotificationById")
        return FaultNotification.model_validate(item)

from raksa.client import EstateAppClient, EstateAppError
from raksa.config import resolve_token, resolve_condo_id, resolve_base_url, RaksaConfigError
from raksa.models import YAMLCase

app = typer.Typer(help="Shareholder renovation notifications", no_args_is_help=True)


def _get_client() -> EstateAppClient:
    try:
        token = resolve_token()
    except RaksaConfigError as e:
        typer.echo(f"Error: {e}", err=True)
        raise typer.Exit(1)
    return EstateAppClient(token=token, base_url=resolve_base_url())


def _get_condo_id(condo_id: str | None) -> str:
    resolved = condo_id or resolve_condo_id()
    if not resolved:
        typer.echo("Error: --condo-id required (or set RAKSA_CONDO_ID)", err=True)
        raise typer.Exit(1)
    return resolved


def load_renovation_cases(yaml_dir: Path) -> list[YAMLCase]:
    cases = []
    for yaml_file in sorted(yaml_dir.rglob("*.yaml")):
        if yaml_file.name.endswith("_chat.yaml"):
            continue
        try:
            data = yaml.safe_load(yaml_file.read_text())
            case = YAMLCase.model_validate(data)
        except Exception as e:
            typer.echo(f"Warning: skipping {yaml_file}: {e}", err=True)
            continue
        if case.is_renovation:
            cases.append(case)
    return cases


@app.command("list")
def list_renovations(condo_id: str = typer.Option(None, "--condo-id", help="Condominium ID")):
    """List renovation notifications for a condominium."""
    client = _get_client()
    cid = _get_condo_id(condo_id)
    try:
        renovations = client.list_renovations(cid)
    except EstateAppError as e:
        typer.echo(f"API error: {e}", err=True)
        raise typer.Exit(1)

    if not renovations:
        typer.echo("No renovations found.")
        return

    for r in renovations:
        srw = r.shareholder_renovation_work
        address = srw.apartment_address if srw else "N/A"
        typer.echo(f"  {r.id}  {r.status or 'N/A':12s}  {address}  {r.title or ''}")


@app.command("get")
def get_renovation(gig_id: str = typer.Argument(help="Renovation gig ID")):
    """Show details of a specific renovation notification."""
    client = _get_client()
    try:
        r = client.get_renovation(gig_id)
    except EstateAppError as e:
        typer.echo(f"API error: {e}", err=True)
        raise typer.Exit(1)

    typer.echo(f"ID:      {r.id}")
    typer.echo(f"Title:   {r.title}")
    typer.echo(f"Status:  {r.status}")
    typer.echo(f"Dates:   {r.start_date} -> {r.end_date}")
    srw = r.shareholder_renovation_work
    if srw:
        typer.echo(f"Address: {srw.apartment_address}")
        typer.echo(f"Work:    {(srw.work_description or '')[:200]}")
        if srw.informant:
            typer.echo(f"Contact: {srw.informant.name} ({srw.informant.email}, {srw.informant.phone})")
        active_jobs = {k: v for k, v in srw.chosen_jobs.model_dump(by_alias=True).items() if v}
        if active_jobs:
            typer.echo(f"Jobs:    {', '.join(active_jobs.keys())}")


@app.command("import")
def import_renovations(
    yaml_dir: Path = typer.Argument(help="Path to hive YAML directory"),
    condo_id: str = typer.Option(None, "--condo-id", help="Condominium ID"),
    submit: bool = typer.Option(False, "--submit", help="Actually submit (default is dry-run)"),
):
    """Import renovation cases from hive-structured YAML directory."""
    cid = _get_condo_id(condo_id)
    cases = load_renovation_cases(yaml_dir)

    if not cases:
        typer.echo("No renovation cases found.")
        return

    typer.echo(f"Found {len(cases)} renovation case(s)\n")

    for case in cases:
        space = case.space.name if case.space else "?"
        timing = case.estimated_timing or "no dates"
        job_summary = case._job_summary()
        typer.echo(f"  [{case.title}]  {space:6s}  {timing:30s}  {job_summary}")

    if not submit:
        typer.echo(f"\nDry run -- {len(cases)} case(s) would be submitted. Use --submit to send.")
        return

    client = _get_client()
    ok = 0
    fail = 0
    for case in cases:
        try:
            gig_input = case.to_renovation_input(condo_id=cid)
            gig_id = client.create_renovation(gig_input)
            typer.echo(f"  OK  [{case.title}] -> {gig_id}")
            ok += 1
        except (EstateAppError, Exception) as e:
            typer.echo(f"  FAIL [{case.title}]: {e}", err=True)
            fail += 1

    typer.echo(f"\nDone: {ok} submitted, {fail} failed")
```
